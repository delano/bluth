    def enqueue(data={},q=nil)
      q = self.queue(q)
      gob = Gob.create generate_id(data), self, data
      gob.current_queue = q.name
      gob.created
      gob.attempts = 0
      gob.save
      Familia.ld "ENQUEUING: #{self} #{gob.jobid.short} to #{q}"
      q << gob.jobid
      gob
    end
